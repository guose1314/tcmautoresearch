1. Get the application URL by running these commands:
{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "tcmautoresearch.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  echo "Run 'kubectl get svc --namespace {{ .Release.Namespace }} {{ include "tcmautoresearch.fullname" . }} -w' to get external IP"
{{- else }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "tcmautoresearch.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8000:{{ .Values.service.targetPort }}
  echo "Open http://127.0.0.1:8000/liveness"
  echo "Open http://127.0.0.1:8000/readiness"
{{- end }}

2. Database migration automation:
{{- if and .Values.migrations.enabled .Values.migrations.useHook }}
  A Helm migration hook is enabled and will run automatically before install/upgrade:
  alembic -x environment=production upgrade head
{{- else if .Values.migrations.enabled }}
  A regular migration Job resource is enabled.
  Inspect it with:
  kubectl get job --namespace {{ .Release.Namespace }} {{ printf "%s-db-migrate" (include "tcmautoresearch.fullname" .) }}
{{- else }}
  Automatic migrations are disabled.
  Re-run your Helm install/upgrade with:
  --set migrations.enabled=true

  For one-off troubleshooting, run:
  kubectl exec --namespace {{ .Release.Namespace }} deploy/{{ include "tcmautoresearch.fullname" . }} -- alembic -x environment=production upgrade head
{{- end }}
